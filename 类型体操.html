# never

never代表该类型的值永远不可能在运行时出现 never是所有类型的子类型，never可以赋值给所有类型

  • 函数：函数的终点不可执行
  • 变量：不可能存在该变量
// 函数
function error(message: string): never {
    throw new Error(message);
}
// 变量
interface Foo {
    type: 'foo';
    name: string;
}
interface Bar {
    type: 'bar';
    age: number;
}
type All = Foo | Bar;
function handleValue(val: All) {
  switch (val.type) {
    case 'foo':
      // 明确到 Foo
      break;
    case 'bar':
      // 明确到 Bar
      break;
    default:
      // 明确到never
      const exhaustive: never = val;
      break;
  }
}

# 映射类型

取key转换新类型

type Readonly<T> = {
    readonly [P in keyof T]: T[P]
}
type Partial<T> = {
    [P in keyof T]?: T[P]
}
type MyMapping<T> =
  T extends string|number ? 'Deny Visit' :
  {
    [K in keyof T]: MyMapping<T[K]>
  }
